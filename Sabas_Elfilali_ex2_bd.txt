A) Création des tables
----------------------

CREATE TABLE PIECE(
    
        NOP number(3),

            DESIGNATION varchar2(30),

            COULEUR varchar2(30),
    
        POIDS VARCHAR2(23)

    );



CREATE TABLE SERVICE(
           
     		NOS NUMBER(4),
             
 	       INTITULE varchar2(30),
              
  	     LOCALISATION VARCHAR2(40)

);


CREATE TABLE ORDRE(
       
      		  NOP NUMBER(5),
           
 	          NOS NUMBER(5),
              
 		  QUANTITE NUMBER(5)

);


CREATE TABLE NOMENCLATURE (
 
	            NOPA NUMBER(5),
       
	            NOPC NUMBER(5),
          
   		   QUANTITE NUMBER(3)


);
